# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def mergeKLists(self, lists: List[ListNode]) -> ListNode:
        self.nodes_list = []
        for lst_node in lists:
            while lst_node != None:
                self.nodes_list.append(lst_node.val)
                lst_node= lst_node.next
        
        self.nodes_list.sort()
        node = ListNode(0)
        head =  None
        first =True
        for num in self.nodes_list:
            node.next = ListNode(num)
            node = node.next
            if first:
                head = node
                first = False
        return head
